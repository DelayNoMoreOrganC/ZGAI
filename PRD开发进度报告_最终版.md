## 📝 技术实现细节

### 待办事项高亮规则实现
```javascript
// 判断是否逾期
const isOverdue = (deadline) => {
  return new Date(deadline) < new Date()
}

// 获取待办样式类
const getTodoClass = (todo) => {
  if (todo.completed) return 'todo-completed'
  if (isOverdue(todo.deadline)) return 'todo-overdue'      // 逾期：红色背景+置顶
  const daysUntilDeadline = Math.ceil((new Date(todo.deadline) - new Date()) / (1000 * 60 * 60 * 24))
  if (daysUntilDeadline <= 3) return 'todo-urgent'        // 3天内：红色文字
  if (daysUntilDeadline <= 7) return 'todo-warning'       // 7天内：橙色文字
  return ''
}
```
